#### 2. –û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è –ø–ª–æ—Ç–Ω–æ—Å—Ç–∏ —Ç–æ—á–µ–∫

**–°–ª–∏—à–∫–æ–º –ø–ª–æ—Ç–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ** (–±–æ–ª–µ–µ 100 000 —Ç–æ—á–µ–∫):

```python
from scipy.spatial import cKDTree

# –ü—Ä–æ—Ä–µ–∂–∏–≤–∞–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö
def thin_points(coords, min_distance=1.0):
    tree = cKDTree(coords[:, :2])  # XY —Ç–æ–ª—å–∫–æ
    keep = [0]
    
    for i in range(1, len(coords)):
        distances, _ = tree.query(coords[i, :2], k=2)
        if distances[1] >= min_distance:
            keep.append(i)
    
    return coords[keep]

# –ü—Ä–∏–º–µ–Ω–µ–Ω–∏–µ
coords = df[['X', 'Y', 'Z']].values
coords_thinned = thin_points(coords, min_distance=2.0)
print(f"–ü—Ä–æ—Ä–µ–∂–µ–Ω–æ —Å {len(coords)} –¥–æ {len(coords_thinned)} —Ç–æ—á–µ–∫")
```

**–ù–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –¥–∞–Ω–Ω—ã—Ö** (–º–µ–Ω–µ–µ 100 —Ç–æ—á–µ–∫):

# WGS84 (lat/lon) ‚Üí UTM zone 37N (–ø—Ä–∏–º–µ—Ä –¥–ª—è –ú–æ—Å–∫–≤—ã)

#### 4. –§–æ—Ä–º–∏—Ä–æ–≤–∞–Ω–∏–µ —Ñ–∏–Ω–∞–ª—å–Ω–æ–≥–æ —Ñ–∞–π–ª–∞

**–ü—Ä–∏–º–µ—Ä —Å–∫—Ä–∏–ø—Ç–∞ –ø–æ–¥–≥–æ—Ç–æ–≤–∫–∏:**

```python
import pandas as pd
import numpy as np

def prepare_data_file(input_file, output_file):
    # –ó–∞–≥—Ä—É–∑–∫–∞
    df = pd.read_csv(input_file, sep='\s+', comment='#', 
                     names=['X', 'Y', 'Z'])
    
    # –û—á–∏—Å—Ç–∫–∞
    df = df.dropna()  # –£–¥–∞–ª–µ–Ω–∏–µ –ø—É—Å—Ç—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π
    df = df.drop_duplicates(subset=['X', 'Y'])  # –£–¥–∞–ª–µ–Ω–∏–µ –¥—É–±–ª–∏–∫–∞—Ç–æ–≤
    
    # –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è –≤—ã–±—Ä–æ—Å–æ–≤
    z_mean, z_std = df['Z'].mean(), df['Z'].std()
    df = df[np.abs(df['Z'] - z_mean) <= 3 * z_std]
    
    # –°–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞ (–æ–ø—Ü–∏–æ–Ω–∞–ª—å–Ω–æ, –¥–ª—è –æ–ø—Ç–∏–º–∏–∑–∞—Ü–∏–∏)
    df = df.sort_values(['X', 'Y'])
    
    # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ
    with open(output_file, 'w') as f:
        f.write('# Prepared geodata file\n')
        f.write('# Points: {}\n'.format(len(df)))
        f.write('# Z range: [{:.2f}, {:.2f}]\n'.format(
            df['Z'].min(), df['Z'].max()))
        f.write('X Y Z\n')
    
    df.to_csv(output_file, sep=' ', index=False, mode='a', header=False)
    print(f"–ü–æ–¥–≥–æ—Ç–æ–≤–ª–µ–Ω–æ {len(df)} —Ç–æ—á–µ–∫ ‚Üí {output_file}")

# –ò—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
prepare_data_file('raw_data.txt', 'prepared_data.txt')
```
